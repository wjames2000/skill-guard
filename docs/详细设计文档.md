```go
// Rule 定义一条检测规则
type Rule struct {
    ID          string   `yaml:"id" json:"id"`
    Name        string   `yaml:"name" json:"name"`
    Severity    string   `yaml:"severity" json:"severity"`
    Description string   `yaml:"description" json:"description"`
    Pattern     string   `yaml:"pattern" json:"pattern"`
    Keywords    []string `yaml:"keywords,omitempty" json:"keywords,omitempty"`
    FileTypes   []string `yaml:"file_types,omitempty" json:"file_types,omitempty"`
    Ref         string   `yaml:"ref,omitempty" json:"ref,omitempty"`

    // 内部字段（运行时编译）
    compiled *regexp.Regexp
}

// Compile 预编译正则表达式
func (r *Rule) Compile() error {
    if r.compiled != nil {
        return nil
    }
    compiled, err := regexp.Compile(r.Pattern)
    if err != nil {
        return fmt.Errorf("rule %s: invalid pattern: %w", r.ID, err)
    }
    r.compiled = compiled
    return nil
}
```

### 3.3 scanner 模块 — 扫描编排器

**文件：** `internal/scanner/scanner.go`、`worker.go`

**职责：**
- 协调完整的扫描流程
- 管理 Worker Pool
- 通过 Channel 实现生产者-消费者模式

**核心逻辑：**

```go
// internal/scanner/scanner.go
func Scan(cfg *config.Config) (*report.ScanReport, error) {
    // 1. 发现文件
    files, err := file.Discover(cfg.Paths, &file.Opts{
        Ignore:     cfg.Ignore,
        ExtInclude: cfg.ExtInclude,
        ExtExclude: cfg.ExtExclude,
        MaxSize:    cfg.MaxSize,
    })
    if err != nil {
        return nil, err
    }

    // 2. 初始化规则引擎
    eng, err := engine.New(cfg.RulesFile, cfg.DisableBuiltin)
    if err != nil {
        return nil, err
    }

    // 3. 并发扫描
    totalFiles := len(files)
    numWorkers := cfg.Concurrency
    if numWorkers <= 0 {
        numWorkers = runtime.NumCPU()
    }

    jobs := make(chan *file.FileTarget, numWorkers)
    results := make(chan []*engine.MatchResult, numWorkers)

    var wg sync.WaitGroup
    for i := 0; i < numWorkers; i++ {
        wg.Add(1)
        go worker(jobs, results, eng, cfg, &wg)
    }

    // 进度统计
    var scanned atomic.Int32
    go func() {
        for _, f := range files {
            jobs <- f
        }
        close(jobs)
    }()

    // 等待完成
    go func() {
        wg.Wait()
        close(results)
    }()

    // 4. 收集结果
    var allResults []*engine.MatchResult
    for res := range results {
        allResults = append(allResults, res...)
        if cfg.Verbose {
            n := scanned.Add(1)
            fmt.Fprintf(os.Stderr, "\r已扫描: %d/%d", n, totalFiles)
        }
    }

    // 5. 构建报告
    return report.Build(allResults, totalFiles), nil
}
```

**Worker 实现：**

```go
// internal/scanner/worker.go
func worker(
    jobs <-chan *file.FileTarget,
    results chan<- []*engine.MatchResult,
    eng *engine.Engine,
    cfg *config.Config,
    wg *sync.WaitGroup,
) {
    defer wg.Done()
    for file := range jobs {
        // 文件预检查
        if !file.CheckEncoding() {
            if cfg.Verbose {
                fmt.Fprintf(os.Stderr, "\n跳过非 UTF-8 文件: %s\n", file.RelPath)
            }
            continue
        }

        // 规则匹配
        matches := eng.Match(file)
        if len(matches) > 0 {
            results <- matches
        }
    }
}
```

### 3.4 engine 模块 — 规则引擎

**文件：** `internal/engine/engine.go`、`rule.go`、`matcher.go`、`loader.go`、`builtin.go`

**职责：**
- 管理规则生命周期（加载、编译、查询）
- 执行匹配（关键字预过滤 → 正则匹配）
- 支持内置规则与自定义规则合并

**Engine 结构：**

```go
// internal/engine/engine.go
type Engine struct {
    rules []*Rule // 已编译的规则列表
    mu    sync.RWMutex
}

// New 创建规则引擎，加载内置规则和自定义规则
func New(rulesFile string, disableBuiltin bool) (*Engine, error) {
    eng := &Engine{}

    if !disableBuiltin {
        eng.rules = append(eng.rules, BuiltinRules()...)
    }

    if rulesFile != "" {
        customRules, err := LoadRulesFile(rulesFile)
        if err != nil {
            return nil, err
        }
        eng.mergeCustom(customRules)
    }

    // 预编译所有规则
    for _, rule := range eng.rules {
        if err := rule.Compile(); err != nil {
            return nil, err
        }
    }

    return eng, nil
}

// Match 对单个文件执行所有规则的匹配
func (e *Engine) Match(file *file.FileTarget) []*MatchResult {
    // 按行读取文件
    lines, err := file.ReadLines()
    if err != nil {
        return nil
    }

    var results []*MatchResult
    for _, rule := range e.rules {
        // 文件类型过滤
        if !rule.MatchesFileType(file.Ext) {
            continue
        }

        // 关键字预过滤
        if !rule.HasKeywords(lines) {
            continue
        }

        // 逐行正则匹配
        for lineNum, line := range lines {
            result := rule.MatchLine(line, file.RelPath, lineNum+1)
            if result != nil {
                results = append(results, result)
                break // 同一规则同一文件只记录一次
            }
        }
    }
    return results
}
```

```go
// internal/engine/matcher.go
// HasKeywords 检查文件是否包含任一预过滤关键字
func (r *Rule) HasKeywords(lines []string) bool {
    if len(r.Keywords) == 0 {
        return true // 无关键字要求，跳过预过滤
    }
    for _, line := range lines {
        for _, kw := range r.Keywords {
            if strings.Contains(line, kw) {
                return true
            }
        }
    }
    return false
}

// MatchLine 对单行执行正则匹配
func (r *Rule) MatchLine(line string, filePath string, lineNum int) *MatchResult {
    if r.compiled == nil {
        return nil
    }

    // panic 保护
    defer func() {
        if rec := recover(); rec != nil {
            // 记录日志，跳过该匹配
        }
    }()

    loc := r.compiled.FindStringIndex(line)
    if loc == nil {
        return nil
    }

    // 截取匹配内容摘要（前 120 字符）
    content := line
    if len(content) > 120 {
        content = content[:120] + "..."
    }

    return &MatchResult{
        RuleID:      r.ID,
        Severity:    r.Severity,
        FilePath:    filePath,
        LineNumber:  lineNum,
        LineContent: strings.TrimSpace(content),
        MatchType:   "regex",
    }
}

// MatchesFileType 检查文件扩展名是否在规则的 FileTypes 范围内
func (r *Rule) MatchesFileType(ext string) bool {
    if len(r.FileTypes) == 0 {
        return true // 未限定文件类型
    }
    for _, ft := range r.FileTypes {
        if strings.EqualFold(ext, ft) {
            return true
        }
    }
    return false
}
```

```go
// internal/engine/loader.go
func LoadRulesFile(path string) ([]*Rule, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, fmt.Errorf("读取规则文件失败: %w", err)
    }

    // 根据扩展名选择解析方式
    var rules []*Rule
    ext := strings.ToLower(filepath.Ext(path))
    switch ext {
    case ".yaml", ".yml":
        err = yaml.Unmarshal(data, &rules)
    case ".json":
        err = json.Unmarshal(data, &rules)
    default:
        return nil, fmt.Errorf("不支持的规则文件格式: %s", ext)
    }
    if err != nil {
        return nil, fmt.Errorf("规则文件解析失败: %w", err)
    }

    // 验证必填字段
    for _, rule := range rules {
        if rule.ID == "" || rule.Pattern == "" || rule.Severity == "" {
            return nil, fmt.Errorf("规则缺少必填字段 (id/pattern/severity)")
        }
    }

    return rules, nil
}

// mergeCustom 合并自定义规则（覆盖或追加）
func (e *Engine) mergeCustom(custom []*Rule) {
    existing := make(map[string]int)
    for i, r := range e.rules {
        existing[r.ID] = i
    }

    for _, cr := range custom {
        if idx, ok := existing[cr.ID]; ok {
            e.rules[idx] = cr // 覆盖
        } else {
            e.rules = append(e.rules, cr) // 追加
        }
    }
}
```

### 3.5 file 模块 — 文件发现与读取

**Discover 实现：**

```go
// internal/file/discover.go
type Opts struct {
    Ignore     []string
    ExtInclude []string
    ExtExclude []string
    MaxSize    int64
}

// 默认忽略路径
var DefaultIgnorePatterns = []string{
    ".git", "node_modules", "vendor",
    "__pycache__", ".svn", ".hg",
    "*.exe", "*.dll", "*.so", "*.dylib",
    "*.jpg", "*.png", "*.gif", "*.ico",
    "*.zip", "*.tar", "*.gz",
}

func Discover(roots []string, opts *Opts) ([]*FileTarget, error) {
    ignoreMatchers := buildIgnoreMatchers(append(DefaultIgnorePatterns, opts.Ignore...))

    var files []*FileTarget
    for _, root := range roots {
        err := filepath.WalkDir(root, func(path string, d os.DirEntry, err error) error {
            if err != nil {
                return nil // 跳过无法访问的路径
            }

            // 忽略目录
            if d.IsDir() {
                if isIgnored(path, ignoreMatchers) {
                    return filepath.SkipDir
                }
                return nil
            }

            // 忽略文件
            if isIgnored(path, ignoreMatchers) {
                return nil
            }

            info, err := d.Info()
            if err != nil {
                return nil
            }

            ext := strings.ToLower(filepath.Ext(path))

            // 扩展名包含过滤
            if len(opts.ExtInclude) > 0 && !matchExt(ext, opts.ExtInclude) {
                return nil
            }
            // 扩展名排除过滤
            if len(opts.ExtExclude) > 0 && matchExt(ext, opts.ExtExclude) {
                return nil
            }

            // 文件大小检查
            if opts.MaxSize > 0 && info.Size() > opts.MaxSize {
                if opts.Verbose {
                    fmt.Fprintf(os.Stderr, "跳过超限文件: %s (%d bytes)\n", path, info.Size())
                }
                return nil
            }

            relPath := getRelPath(root, path)
            files = append(files, &FileTarget{
                Path:    path,
                RelPath: relPath,
                Size:    info.Size(),
                Ext:     ext,
            })
            return nil
        })
        if err != nil {
            return nil, fmt.Errorf("遍历目录失败 %s: %w", root, err)
        }
    }
    return files, nil
}
```

```go
const DefaultBufferSize = 64 * 1024 // 64KB

func (f *FileTarget) ReadLines() ([]string, error) {
    file, err := os.Open(f.Path)
    if err != nil {
        return nil, err
    }
    defer file.Close()

    var lines []string
    scanner := bufio.NewScanner(file)
    scanner.Buffer(make([]byte, DefaultBufferSize), DefaultBufferSize)

    for scanner.Scan() {
        lines = append(lines, scanner.Text())
    }
    return lines, scanner.Err()
}
```

### 3.6 report 模块 — 结果聚合

```go
// internal/report/report.go
func Build(results []*MatchResult, totalFiles int) *ScanReport {
    results = Aggregate(results)
    return &ScanReport{
        ScanTime:    time.Now().Format(time.RFC3339),
        Duration:    "", // 由调用方设置
        TotalFiles:  totalFiles,
        TotalIssues: len(results),
        Results:     results,

    }
}

// internal/report/aggregate.go
var severityOrder = map[string]int{
    "Critical": 0,
    "High":     1,
    "Medium":   2,
    "Low":      3,
}

func Aggregate(results []*MatchResult) []*MatchResult {
    // 1. 去重（同一规则 + 同一文件只保留一条）
    seen := make(map[string]bool)
    var unique []*MatchResult
    for _, r := range results {
        key := r.RuleID + ":" + r.FilePath
        if seen[key] {
            continue
        }
        seen[key] = true
        unique = append(unique, r)
    }

    // 2. 按严重级别排序
    sort.Slice(unique, func(i, j int) bool {
        return severityOrder[unique[i].Severity] < severityOrder[unique[j].Severity]
    })

    return unique
}
```

```go
    fmt.Fprintf(w, "╔════════════════════════════════════════════════════════╗\n")
    fmt.Fprintf(w, "║  skill-guard 扫描报告                                  ║\n")
    fmt.Fprintf(w, "║  扫描时间: %s  ║\n", report.ScanTime)
    fmt.Fprintf(w, "║  扫描文件数: %-41d ║\n", report.TotalFiles)
    fmt.Fprintf(w, "║  发现风险: %-43d ║\n", report.TotalIssues)
    fmt.Fprintf(w, "╚════════════════════════════════════════════════════════╝\n\n")

    if len(report.Results) == 0 {
        fmt.Fprintln(w, "✓ 未发现安全风险")
        return nil
    }

    // 2. 逐条输出
    for _, result := range report.Results {
        color := severityColor(result.Severity)
        fmt.Fprintf(w, "%s [%s] %s%s\n",
            color, result.Severity, result.RuleID, result.RuleName)
        fmt.Fprintf(w, "  → %s:%d\n", result.FilePath, result.LineNumber)
        fmt.Fprintf(w, "    %s\n", result.LineContent)
        fmt.Fprint(w, resetColor)
    }

    return nil
}
```

## 4. 并发设计

### 4.1 Worker Pool 模式

```
                     ┌──────────┐
                     │ 文件列表   │
                     └────┬─────┘
                          │
                    ┌─────▼──────┐
                    │  jobs chan  │ 缓冲大小 = numWorkers
                    └─────┬──────┘
                          │
          ┌───────────────┼───────────────┐
          │               │               │
    ┌─────▼─────┐  ┌─────▼─────┐  ┌─────▼─────┐
    │ Worker 1  │  │ Worker 2  │  │ Worker N  │
    │ goroutine │  │ goroutine │  │ goroutine │
    └─────┬─────┘  └─────┬─────┘  └─────┬─────┘
          │               │               │
          └───────────────┼───────────────┘
                          │
                    ┌─────▼──────┐
                    │ results chan│
                    └─────┬──────┘
                          │
                    ┌─────▼─────┐
                    │  结果聚合   │
                    └───────────┘
```

### 4.2 并发安全

- **规则引擎：** 只读不写，多个 goroutine 可安全共享（编译后的 `*regexp.Regexp` 是并发安全的）
- **文件读取：** 每个 goroutine 独立打开文件，无共享文件句柄
- **结果收集：** 通过 channel 传递，无共享内存
- **进度统计：** 使用 `sync/atomic` 计数器

## 5. 错误处理设计

### 5.1 错误分类

```go
// 致命错误：终止扫描
var (
    ErrDirNotFound      = errors.New("目录不存在")
    ErrConfigParseError = errors.New("配置文件格式错误")
    ErrRuleParseError   = errors.New("规则文件格式错误")
    ErrInvalidRegex     = errors.New("无效的正则表达式")
)

// 非致命错误：跳过继续
var (
    ErrFileUnreadable  = errors.New("文件无法读取")
    ErrFileTooLarge    = errors.New("文件超过大小限制")
    ErrEncodingInvalid = errors.New("非 UTF-8 编码")
)
```

### 5.2 错误处理策略

| 错误来源 | 处理方式 | 是否终止 |
|---------|---------|---------|
| 目录路径无效 | 输出 ERROR，终止 | 是 |
| 配置文件解析失败 | 输出 ERROR，终止 | 是 |
| 自定义规则语法错误 | 输出 ERROR，终止 | 是 |
| 正则编译失败 | 输出 ERROR，终止 | 是 |
| 单文件读取失败 | 跳过，打印 Warning | 否 |
| 文件编码问题 | 跳过，打印 Warning | 否 |
| 文件超限 | 跳过，打印 Warning | 否 |
| 正则匹配 panic | recover，跳过单条规则 | 否 |

```go
// rules/rules_test.go
func TestRuleSKL001(t *testing.T) {
    tests := []struct {
        name     string
        input    string
        wantMatch bool
    }{
        {"标准 AWS Key", "AKIAIOSFODNN7EXAMPLE", true},
        {"小写格式", "akiaiosfodnn7example", true},
        {"SQL 中的 Key", "INSERT INTO keys VALUES ('AKIAIOSFODNN7EXAMPLE')", true},
        {"JSON 中的 Key", `"aws_key": "AKIAIOSFODNN7EXAMPLE"`, true},
        {"YAML 中的 Key", "access_key: AKIAIOSFODNN7EXAMPLE", true},
        {"短字符串", "AKIA123", false},
        {"注释引用", "# AKIAIOSFODNN7EXAMPLE is an example", true},
        {"不完整前缀", "AKIBIOSFODNN7EXAMPLE", false},
        {"环境变量引用", "${AWS_ACCESS_KEY_ID}", false},
    }

    rule := &Rule{
        ID:       "SKL-001",
        Pattern:  `(?i)AKIA[A-Z0-9]{16}`,
        Keywords: []string{"AKIA"},
    }
    rule.Compile()

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            result := rule.MatchLine(tt.input, "test.py", 1)
            if tt.wantMatch && result == nil {
                t.Errorf("期望匹配，但未匹配: %s", tt.input)
            }
            if !tt.wantMatch && result != nil {
                t.Errorf("期望不匹配，但匹配了: %s", tt.input)
            }
        })
    }
}
```

### 7.1 构建脚本

```bash
#!/bin/bash
# scripts/build.sh

VERSION=$(git describe --tags --always --dirty 2>/dev/null || echo "dev")
COMMIT=$(git rev-parse --short HEAD 2>/dev/null || echo "unknown")
DATE=$(date -u +'%Y-%m-%dT%H:%M:%SZ')

LDFLAGS="-X main.Version=${VERSION} -X main.Commit=${COMMIT} -X main.Date=${DATE}"

PLATFORMS=(
    "linux/amd64"
    "linux/arm64"
    "darwin/amd64"
    "darwin/arm64"
    "windows/amd64"
)

for PLATFORM in "${PLATFORMS[@]}"; do
    GOOS=${PLATFORM%/*}
    GOARCH=${PLATFORM#*/}
    OUTPUT="skill-guard-${GOOS}-${GOARCH}"
    if [ "$GOOS" = "windows" ]; then
        OUTPUT+=".exe"
    fi
    GOOS=$GOOS GOARCH=$GOARCH go build -ldflags "${LDFLAGS}" -o "dist/${OUTPUT}" .
done
```
